\begin{frame}{Aufstellen eines SEM}
    \begin{itemize}[label={}]
        \item $Z \in \Set{\text{klein}, \text{groß}}$: Größe des Nierensteins
        \item $T \in \Set{A, B}$: Behandlung (Treatment)
        \item $R \in \Set{\text{erfolg}, \text{misserfolg}}$: Behandlungserfolg (Recovery)
    \end{itemize}
